<!-- Fix insane Canvas zoom on static export -->

<script>

document.addEventListener("DOMContentLoaded", () => {

  const canvas = document.querySelector(

    '.obsidian-document[data-type="canvas"] .canvas'

  );


  if (!canvas) return;


  const transform = canvas.style.transform || "";

  const match = transform.match(/scale\(([-0-9.]+)\)/);


  if (!match) return;


  const currentScale = parseFloat(match[1]);


  const MIN_SCALE = 1.0; // force normal zoom


  if (currentScale < MIN_SCALE) {

    canvas.style.transform = transform.replace(

      /scale\(([-0-9.]+)\)/,

      `scale(${MIN_SCALE})`

    );

  }

});

</script>
